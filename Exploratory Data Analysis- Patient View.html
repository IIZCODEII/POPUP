def rename_unnamed(df, label=''):
    for i, cols in enumerate(df.columns.levels):
        cols = cols.tolist()
        for k, col in enumerate(cols):
            if 'Unnamed' in col:
                cols[k] = ' '*k
        df.columns.set_levels(cols,level=i,inplace=True)
